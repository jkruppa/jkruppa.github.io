  mutate(trt = factor(trt, labels = c("ctrl", "A", "B", "C")),

data_long_tbl <- expand_grid(location = 1:2, trt = 1:4, block = 1:4) %>%

         trt = factor(trt, labels = c("ctrl", "A", "B", "C")),
