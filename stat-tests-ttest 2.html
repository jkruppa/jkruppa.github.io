    Rest Formeln

    \[ Pr\left(\left.\cfrac{\bar{y}_1-\bar{y}_2}{s_{y_1,\,y_2} \cdot \sqrt{2/n_g}}\right| \bar{y}_1-\bar{y}_2=0\right) \]

    \[ Pr\left(\left.\cfrac{\Delta_{y_1,\, y_2}}{s_{y_1,\,y_2} \cdot \sqrt{2/n_g}}\right| \Delta_{y_1,\,y_2}=0\right) \]

    \[ Pr\left(\left.T_{\alpha = 5\%}\right| \Delta_{y_1,\,y_2}=0\right) \]

    \[ T = \cfrac{\Delta \cdot n}{s} \]

    \[ \text{Pr}(D | H_0) \]

    \[\begin{align*} T_{calc} &= \cfrac{7.24 - 9.71}{6.45 \cdot \sqrt{\frac{2}{7.5}}} \\ &= \cfrac{-2.47}{6.45 \cdot 0.52} \\ &= \cfrac{-2.47}{3.354} = -0.73 \end{align*}\] \[\begin{align*} H_0: \bar{y}_{dog} &= \bar{y}_{cat} \\ H_A: \bar{y}_{dog} &\neq \bar{y}_{cat} \\ \end{align*}\]
